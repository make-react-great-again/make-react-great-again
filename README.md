## Install

```bash
npm install --save make-react-great-again
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'make-react-great-again'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
